186    /** Convert bitset to a bit string */
187    private static String bitset2string(BitSet b) {
188        StringBuffer sb = new StringBuffer();
189        for (int i = 0; i < b.length(); i++)
190            sb.append(b.get(i) ? "1" : "0");
191        return sb.toString();
192    }

217    /**
218     * Save an XML file
219     * 
220     * @param outFile
221     *            output file
222     * @throws Exception thrown when the save fails
223     */
224    public void save(File outFile) throws Exception {
225        if (outFile == null) {
226            outFile = new File(iOutputFolder, "solution.xml");
227        } else if (outFile.getParentFile() != null) {
228            outFile.getParentFile().mkdirs();
229        }
230        sLogger.debug("Writting XML data to:" + outFile);
231
232        Document document = DocumentHelper.createDocument();
233        document.addComment("Student Sectioning");
234        
235        populate(document);
236
237        FileOutputStream fos = null;
238        try {
239            fos = new FileOutputStream(outFile);
240            (new XMLWriter(fos, OutputFormat.createPrettyPrint())).write(document);
241            fos.flush();
242            fos.close();
243            fos = null;
244        } finally {
245            try {
246                if (fos != null)
247                    fos.close();
248            } catch (IOException e) {
249            }
250        }
251
252        if (iConvertIds)
253            IdConvertor.getInstance().save();
254    }

256    public Document saveDocument() {
257        Document document = DocumentHelper.createDocument();
258        document.addComment("Student Sectioning");
259        
260        populate(document);
261
262        return document;
263    }

265    /**
266     * Fill in all the data into the given document
267     * @param document document to be populated
268     */
269    protected void populate(Document document) {
270        if (iSaveCurrent || iSaveBest) {
271            StringBuffer comments = new StringBuffer("Solution Info:\n");
272            Map<String, String> solutionInfo = (getSolution() == null ? getModel().getExtendedInfo(getAssignment()) : getSolution().getExtendedInfo());
273            for (String key : new TreeSet<String>(solutionInfo.keySet())) {
274                String value = solutionInfo.get(key);
275                comments.append("    " + key + ": " + value + "\n");
276            }
277            document.addComment(comments.toString());
278        }
279
280        Element root = document.addElement("sectioning");
281        root.addAttribute("version", "1.0");
282        root.addAttribute("initiative", getModel().getProperties().getProperty("Data.Initiative"));
283        root.addAttribute("term", getModel().getProperties().getProperty("Data.Term"));
284        root.addAttribute("year", getModel().getProperties().getProperty("Data.Year"));
285        root.addAttribute("created", String.valueOf(new Date()));
286
287        saveOfferings(root);
288
289        saveStudents(root);
290        
291        saveLinkedSections(root);
292        
293        saveTravelTimes(root);
294
295        if (iShowNames) {
296            Progress.getInstance(getModel()).save(root);
297        }
298    }

302     * @param root document root
303     */
304    protected void saveOfferings(Element root) {
305        Element offeringsEl = root.addElement("offerings");
306        for (Offering offering : getModel().getOfferings()) {
307            Element offeringEl = offeringsEl.addElement("offering");
308            saveOffering(offeringEl, offering);
309            saveReservations(offeringEl, offering);
310            saveRestrictions(offeringEl, offering);
311        }
312    }

314    /**
315     * Save given offering
316     * @param offeringEl offering element to be populated
317     * @param offering offering to be saved
318     */
319    protected void saveOffering(Element offeringEl, Offering offering) {
320        offeringEl.addAttribute("id", getId("offering", offering.getId()));
321        if (iShowNames)
322            offeringEl.addAttribute("name", offering.getName());
323        if (offering.isDummy())
324            offeringEl.addAttribute("dummy", "true");
325        for (Course course : offering.getCourses()) {
326            Element courseEl = offeringEl.addElement("course");
327            saveCourse(courseEl, course);
328        }
329        for (Config config : offering.getConfigs()) {
330            Element configEl = offeringEl.addElement("config");
331            saveConfig(configEl, config);
332        }
333    }

340    protected void saveCourse(Element courseEl, Course course) {
341        courseEl.addAttribute("id", getId("course", course.getId()));

346        if (iShowNames && course.getLimit() >= 0)

348        if (iShowNames && course.getProjected() != 0)
349            courseEl.addAttribute("projected", String.valueOf(course.getProjected()));
350        if (iShowNames && course.getCredit() != null)
351            courseEl.addAttribute("credit", course.getCredit());
352        if (course.hasCreditValue())
353            courseEl.addAttribute("credits", course.getCreditValue().toString());
354        if (iShowNames && course.getType() != null)
355            courseEl.addAttribute("type", course.getType());
356        if (iShowNames && course.getTitle() != null)
357            courseEl.addAttribute("title", course.getTitle());
358        if (iShowNames && course.getNote() != null)
359            courseEl.addAttribute("note", course.getNote());
360        
361    }

385        }

402            if (subpart.hasCreditValue())
403                subpartEl.addAttribute("credits", subpart.getCreditValue().toString());
404        }

406            subpartEl.addAttribute("allowOverlap", "true");
407        for (Section section : subpart.getSections()) {
408            Element sectionEl = subpartEl.addElement("section");

413    /**
414     * Save section
415     * @param sectionEl section element to be populated
416     * @param section section to be saved
417     */
418    protected void saveSection(Element sectionEl, Section section) {
419        sectionEl.addAttribute("id", getId("section", section.getId()));
420        sectionEl.addAttribute("limit", String.valueOf(section.getLimit()));
421        if (section.isCancelled())
422            sectionEl.addAttribute("cancelled", "true");
423        if (!section.isEnabled())
424            sectionEl.addAttribute("enabled", "false");
425        if (section.isOnline())
426            sectionEl.addAttribute("online", "true");
427        if (section.isPast())
428            sectionEl.addAttribute("past", "true");
429        if (iShowNames && section.getNameByCourse() != null)
430            for (Map.Entry<Long, String> entry: section.getNameByCourse().entrySet())
431                sectionEl.addElement("cname").addAttribute("id", getId("course", entry.getKey())).setText(entry.getValue());
432        if (section.getParent() != null)
433            sectionEl.addAttribute("parent", getId("section", section.getParent().getId()));
434        if (section.hasInstructors()) {
435            for (Instructor instructor: section.getInstructors()) {
436                Element instructorEl = sectionEl.addElement("instructor");
437                instructorEl.addAttribute("id", getId("instructor", instructor.getId()));
438                if (iShowNames && instructor.getName() != null)
439                    instructorEl.addAttribute("name", instructor.getName());
440                if (iShowNames && instructor.getExternalId() != null)
441                    instructorEl.addAttribute("externalId", instructor.getExternalId());
442                if (iShowNames && instructor.getEmail() != null)
443                    instructorEl.addAttribute("email", instructor.getExternalId());
444            }
445        }
446        if (iShowNames)
447            sectionEl.addAttribute("name", section.getName());
448        if (section.getPlacement() != null) {
449            TimeLocation tl = section.getPlacement().getTimeLocation();
450            if (tl != null) {
451                Element timeLocationEl = sectionEl.addElement("time");
452                timeLocationEl.addAttribute("days", sDF[7].format(Long.parseLong(Integer
453                        .toBinaryString(tl.getDayCode()))));
454                timeLocationEl.addAttribute("start", String.valueOf(tl.getStartSlot()));
455                timeLocationEl.addAttribute("length", String.valueOf(tl.getLength()));
456                if (tl.getBreakTime() != 0)
457                    timeLocationEl.addAttribute("breakTime", String.valueOf(tl.getBreakTime()));
458                if (iShowNames && tl.getTimePatternId() != null)
459                    timeLocationEl.addAttribute("pattern", getId("timePattern", tl.getTimePatternId()));
460                if (iShowNames && tl.getDatePatternId() != null)
461                    timeLocationEl.addAttribute("datePattern", tl.getDatePatternId().toString());
462                if (iShowNames && tl.getDatePatternName() != null
463                        && tl.getDatePatternName().length() > 0)
464                    timeLocationEl.addAttribute("datePatternName", tl.getDatePatternName());
465                timeLocationEl.addAttribute("dates", bitset2string(tl.getWeekCode()));
466                if (iShowNames)
467                    timeLocationEl.setText(tl.getLongName(true));
468            }
469            for (RoomLocation rl : section.getRooms()) {
470                Element roomLocationEl = sectionEl.addElement("room");
471                roomLocationEl.addAttribute("id", getId("room", rl.getId()));
472                if (iShowNames && rl.getBuildingId() != null)
473                    roomLocationEl.addAttribute("building", getId("building", rl.getBuildingId()));
474                if (iShowNames && rl.getName() != null)
475                    roomLocationEl.addAttribute("name", rl.getName());
476                roomLocationEl.addAttribute("capacity", String.valueOf(rl.getRoomSize()));
477                if (rl.getPosX() != null && rl.getPosY() != null)
478                    roomLocationEl.addAttribute("location", rl.getPosX() + "," + rl.getPosY());
479                if (rl.getIgnoreTooFar())
480                    roomLocationEl.addAttribute("ignoreTooFar", "true");
481            }
482        }
483        if (iSaveOnlineSectioningInfo) {
484            if (section.getSpaceHeld() != 0.0)
485                sectionEl.addAttribute("hold", sStudentWeightFormat.format(section.getSpaceHeld()));
486            if (section.getSpaceExpected() != 0.0)
487                sectionEl.addAttribute("expect", sStudentWeightFormat
488                        .format(section.getSpaceExpected()));
489        }
490        if (section.getIgnoreConflictWithSectionIds() != null && !section.getIgnoreConflictWithSectionIds().isEmpty()) {
491            Element ignoreEl = sectionEl.addElement("no-conflicts");
492            for (Long sectionId: section.getIgnoreConflictWithSectionIds())
493                ignoreEl.addElement("section").addAttribute("id", getId("section", sectionId));
494        }
495    }

510    /**
511     * Save reservation
512     * @param reservationEl reservation element to be populated
513     * @param reservation reservation to be saved
514     */
515    protected void saveReservation(Element reservationEl, Reservation reservation) {
516        reservationEl.addAttribute("id", getId("reservation", reservation.getId()));
517        reservationEl.addAttribute("expired", reservation.isExpired() ? "true" : "false");
518        if (reservation instanceof LearningCommunityReservation) {
519            LearningCommunityReservation lc = (LearningCommunityReservation)reservation;
520            reservationEl.addAttribute("type", "lc");
521            for (Long studentId: lc.getStudentIds())
522                reservationEl.addElement("student").addAttribute("id", getId("student", studentId));
523            if (lc.getReservationLimit() >= 0.0)
524                reservationEl.addAttribute("limit", String.valueOf(lc.getReservationLimit()));
525            reservationEl.addAttribute("course", getId("course",lc.getCourse().getId()));
526        } else if (reservation instanceof GroupReservation) {
527            GroupReservation gr = (GroupReservation)reservation;
528            reservationEl.addAttribute("type", "group");
529            for (Long studentId: gr.getStudentIds())
530                reservationEl.addElement("student").addAttribute("id", getId("student", studentId));
531            if (gr.getReservationLimit() >= 0.0)
532                reservationEl.addAttribute("limit", String.valueOf(gr.getReservationLimit()));
533        } else if (reservation instanceof ReservationOverride) {
534            reservationEl.addAttribute("type", "override");
535            ReservationOverride o = (ReservationOverride)reservation;
536            for (Long studentId: o.getStudentIds())
537                reservationEl.addElement("student").addAttribute("id", getId("student", studentId));
538        } else if (reservation instanceof IndividualReservation) {
539            reservationEl.addAttribute("type", "individual");
540            for (Long studentId: ((IndividualReservation)reservation).getStudentIds())
541                reservationEl.addElement("student").addAttribute("id", getId("student", studentId));
542        } else if (reservation instanceof CurriculumReservation) {
543            reservationEl.addAttribute("type", (reservation instanceof CurriculumOverride ? "curriculum-override" : "curriculum"));
544            CurriculumReservation cr = (CurriculumReservation)reservation;
545            if (cr.getReservationLimit() >= 0.0)
546                reservationEl.addAttribute("limit", String.valueOf(cr.getReservationLimit()));
547            if (cr.getAcademicAreas().size() == 1)
548                reservationEl.addAttribute("area", cr.getAcademicAreas().iterator().next());
549            else {
550                for (String area: cr.getAcademicAreas())
551                    reservationEl.addElement("area").addAttribute("code", area);
552            }
553            for (String clasf: cr.getClassifications())
554                reservationEl.addElement("classification").addAttribute("code", clasf);
555            for (String major: cr.getMajors()) {
556                Element majorEl = reservationEl.addElement("major").addAttribute("code", major);
557                Set<String> concentrations = cr.getConcentrations(major);
558                if (concentrations != null)
559                    for (String conc: concentrations)
560                        majorEl.addElement("concentration").addAttribute("code", conc);
561            }
562            for (String minor: cr.getMinors())
563                reservationEl.addElement("minor").addAttribute("code", minor);
564        } else if (reservation instanceof CourseReservation) {
565            reservationEl.addAttribute("type", "course");
566            CourseReservation cr = (CourseReservation)reservation;
567            reservationEl.addAttribute("course", getId("course",cr.getCourse().getId()));
568        } else if (reservation instanceof DummyReservation) {
569            reservationEl.addAttribute("type", "dummy");
570        } else if (reservation instanceof UniversalOverride) {
571            reservationEl.addAttribute("type", "universal");
572            UniversalOverride ur = (UniversalOverride)reservation;
573            if (ur.getFilter() != null)
574                reservationEl.addAttribute("filter", ur.getFilter());
575            reservationEl.addAttribute("override", ur.isOverride() ? "true" : "false");

609     * Save restriction
610     * @param restrictionEl restriction element to be populated
611     * @param restriction restriction to be saved
612     */
613    protected void saveRestriction(Element restrictionEl, Restriction restriction) {
614        restrictionEl.addAttribute("id", getId("restriction", restriction.getId()));
615        if (restriction instanceof IndividualRestriction) {
616            restrictionEl.addAttribute("type", "individual");
617            for (Long studentId: ((IndividualRestriction)restriction).getStudentIds())
618                restrictionEl.addElement("student").addAttribute("id", getId("student", studentId));
619        } else if (restriction instanceof CurriculumRestriction) {
620            restrictionEl.addAttribute("type", "curriculum");
621            CurriculumRestriction cr = (CurriculumRestriction)restriction;
622            if (cr.getAcademicAreas().size() == 1)
623                restrictionEl.addAttribute("area", cr.getAcademicAreas().iterator().next());
624            else {
625                for (String area: cr.getAcademicAreas())
626                    restrictionEl.addElement("area").addAttribute("code", area);
627            }
628            for (String clasf: cr.getClassifications())
629                restrictionEl.addElement("classification").addAttribute("code", clasf);
630            for (String major: cr.getMajors()) {
631                Element majorEl = restrictionEl.addElement("major").addAttribute("code", major);
632                Set<String> concentrations = cr.getConcentrations(major);
633                if (concentrations != null)
634                    for (String conc: concentrations)
635                        majorEl.addElement("concentration").addAttribute("code", conc);
636            }
637            for (String minor: cr.getMinors())
638                restrictionEl.addElement("minor").addAttribute("code", minor);
639        } else if (restriction instanceof CourseRestriction) {
640            restrictionEl.addAttribute("type", "course");
641            CourseRestriction cr = (CourseRestriction)restriction;
642            restrictionEl.addAttribute("course", getId("course",cr.getCourse().getId()));
643        }
644        for (Config config: restriction.getConfigs())
645            restrictionEl.addElement("config").addAttribute("id", getId("config", config.getId()));
646        for (Map.Entry<Subpart, Set<Section>> entry: restriction.getSections().entrySet()) {
647            for (Section section: entry.getValue()) {
648                restrictionEl.addElement("section").addAttribute("id", getId("section", section.getId()));
649            }
650        }
651    }

823        requestEl.addAttribute("id", getId("request", request.getId()));
824        requestEl.addAttribute("priority", String.valueOf(request.getPriority()));
825        if (request.isAlternative())
826            requestEl.addAttribute("alternative", "true");

841        if (iSaveInitial && request.getInitialAssignment() != null) {

843        }

845            requestEl.addElement("current");

848            requestEl.addElement("best");

852    /**
853     * Save course request 
854     * @param requestEl request element to be populated
855     * @param request course request to be saved
856     */
857    protected void saveCourseRequest(Element requestEl, CourseRequest request) {
858        requestEl.addAttribute("id", getId("request", request.getId()));
859        requestEl.addAttribute("priority", String.valueOf(request.getPriority()));
860        if (request.isAlternative())
861            requestEl.addAttribute("alternative", "true");
862        if (request.getWeight() != 1.0)
863            requestEl.addAttribute("weight", sStudentWeightFormat.format(request.getWeight()));
864        requestEl.addAttribute("waitlist", request.isWaitlist() ? "true" : "false");
865        if (request.getRequestPriority() != RequestPriority.Normal)
866            requestEl.addAttribute("importance", request.getRequestPriority().name());
867        if (request.getRequestPriority() == RequestPriority.Critical)
868            requestEl.addAttribute("critical", "true");
869        if (request.getTimeStamp() != null)
870            requestEl.addAttribute("timeStamp", request.getTimeStamp().toString());
871        boolean first = true;
872        for (Course course : request.getCourses()) {
873            if (first)
874                requestEl.addAttribute("course", getId("course", course.getId()));
875            else
876                requestEl.addElement("alternative").addAttribute("course", getId("course", course.getId()));
877            first = false;
878        }
879        for (Choice choice : request.getWaitlistedChoices()) {
880            Element choiceEl = requestEl.addElement("waitlisted");
881            choiceEl.addAttribute("offering", getId("offering", choice.getOffering().getId()));
882            choiceEl.setText(choice.getId());
883        }
884        for (Choice choice : request.getSelectedChoices()) {
885            Element choiceEl = requestEl.addElement("selected");
886            choiceEl.addAttribute("offering", getId("offering", choice.getOffering().getId()));
887            choiceEl.setText(choice.getId());
888        }
889        for (Choice choice : request.getRequiredChoices()) {
890            Element choiceEl = requestEl.addElement("required");
891            choiceEl.addAttribute("offering", getId("offering", choice.getOffering().getId()));
892            choiceEl.setText(choice.getId());
893        }
894        if (iSaveInitial && request.getInitialAssignment() != null) {
895            saveEnrollment(requestEl.addElement("initial"), request.getInitialAssignment());
896        }
897        if (iSaveCurrent && getAssignment().getValue(request) != null) {
898            saveEnrollment(requestEl.addElement("current"), getAssignment().getValue(request));
899        }
900        if (iSaveBest && request.getBestAssignment() != null) {
901            saveEnrollment(requestEl.addElement("best"), request.getBestAssignment());
902        }
903        if (request.isFixed())
904            saveEnrollment(requestEl.addElement("fixed"), request.getFixedValue());
905        for (RequestGroup g: request.getRequestGroups()) {
906            Element groupEl = requestEl.addElement("group").addAttribute("id", getId("group", g.getId())).addAttribute("course", getId("course", g.getCourse().getId()));
907            if (iShowNames)
908                groupEl.addAttribute("name", g.getName());
909        }
910    }
