032 * @param <V> Variable
033 * @param <T> Value
034 **/
035public interface SolverListener<V extends Variable<V, T>, T extends Value<V, T>> {
036
037    /**
038     * A variable was selected
039     * 
040     * @param assignment current assignment
041     * @param iteration
042     *            current iteration
043     * @param variable
044     *            selected variable
045     * @return if false is returned the variable will be discarded (search continues with the next variable selection)
046     */
047    public boolean variableSelected(Assignment<V, T> assignment, long iteration, V variable);
048
049    /**
050     * A value was selected
051     * 
052     * @param assignment current assignment
053     * @param iteration
054     *            current iteration
055     * @param variable
056     *            selected variable
057     * @param value
058     *            selected variable
059     * @return if false is returned the selected value is not assigned
060     */
061    public boolean valueSelected(Assignment<V, T> assignment, long iteration, V variable, T value);
062
063    /**
064     * A neighbour was selected
065     * 
066     * @param assignment current assignment
067     * @param iteration
068     *            current iteration
069     * @param neighbour
070     *            neighbour
071     * @return if false is returned the selected neighbour is not assigned
072     */
073    public boolean neighbourSelected(Assignment<V, T> assignment, long iteration, Neighbour<V, T> neighbour);
074    
075    /**
076     * Called when {@link ParallelSolver} failed to assign the given neighbour
077     * 
078     * @param assignment current assignment
079     * @param iteration
080     *            current iteration
081     * @param neighbour
082     *            neighbour
083     */
084    public void neighbourFailed(Assignment<V, T> assignment, long iteration, Neighbour<V, T> neighbour);
085
086}
