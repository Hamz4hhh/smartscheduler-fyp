032public class Constants {
033        private static Properties sProperties = null;
034        
035        private static Properties getProperties() {
036                if (sProperties == null) {
037                        sProperties = new Properties();
038                        InputStream in = Constants.class.getClassLoader().getResourceAsStream("cpsolver.version");
039                        if (in != null) {
040                                try {
041                                        sProperties.load(in);
042                                        in.close();
043                                } catch (IOException e) {}
044                        }
045                }
046                return sProperties;
047        }
048        
049    /**
050     * Version
051     * @return current solver version
052     */
053    public static String getVersion() {
054        return getProperties().getProperty("project.version", "1.3");
055    }
056
057    /**
058     * Build number
059     * @return current solver build number
060     */
061    public static String getBuildNumber() {
062        return getProperties().getProperty("cpsolver.build_nbr", "?");
063    }
064
065    /**
066     * Release date
067     * @return current solver release date
068     */
069    public static String getReleaseDate() {
070        return getProperties().getProperty("cpsolver.rel_date", "?");
071    }
072}
