037    /** Perform assignment 
038     * @param assignment current assignment
039     * @param iteration current iteration
040     **/
041    public void assign(Assignment<V, T> assignment, long iteration);
042
043    /** Difference in the evaluation function, if this neighnour is assigned.
044     * @param assignment current assignment
045     * @return difference in the solution value when assigned
046     **/
047    public double value(Assignment<V, T> assignment);
048    
049    /** Return assignments to be done.
050     * @return list of assignments (including unassignments) of this neighbour
051     **/
