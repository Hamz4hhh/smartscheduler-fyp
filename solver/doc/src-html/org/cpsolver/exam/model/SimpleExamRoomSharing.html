031public class SimpleExamRoomSharing extends ExamRoomSharing {
032
033    public SimpleExamRoomSharing(Model<Exam, ExamPlacement> model, DataProperties config) {
034        super(model, config);
035    }
036
037    @Override
038    public boolean canShareRoom(Exam x1, Exam x2) {
039        return x1.getLength() == x2.getLength();
040    }
